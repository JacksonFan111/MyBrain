-- Script to look at the actual percentage for each fee schedule
--mySTART Fee Schedule
--Craigs KS Fee Schedule
--CIP Super Fee Schedule
--superSTART Fee Schedule


--FeePackageName = 'Craigs KS Fee Schedule'
--AND FeeName = 'Management Fee'


WITH AssetTypeMapping AS (
    SELECT *
    FROM (VALUES
        (NULL, NULL, 'Cash Call Account'),
        (2, 'FIX', 'Term Deposits & Debentures'),
        (4, 'UTR', 'Unitised'),
        (5, 'SHA', 'Shares'),
        (6, 'GOV', 'Government Bonds'),
        (7, 'BND', 'Bonds & Notes')
    ) AS AssetTypeMap (AssetType, ProductType, TypeName)
),
RankedFeeScales AS (
    SELECT 
        fp.Name AS FeePackageName,
        FORMAT(fp.AuthorisedDate, 'yyyy-MM-dd') AS AuthorisedDate,
        FORMAT(fp.CreatedDate, 'yyyy-MM-dd') AS CreatedDate,
        f.ActualisationFrequency,
        f.CalculationMethod,
        f.Name AS FeeName,
        atm.AssetType,
        COALESCE(atm.TypeName, 'Cash Call Account') AS AssetTypeName,
        fs.EndValue,
        fs.FeeValue as AnnualisedFeeInPercentage,
        ROW_NUMBER() OVER (PARTITION BY fp.Name, atm.TypeName ORDER BY fs.EndValue) AS RowNum
    FROM
        dbo.FeePackage fp
    LEFT JOIN 
        dbo.Fee f ON f.FeePackage_ID = fp.Id
    LEFT JOIN 
        dbo.FeeScale fs ON fs.Fee_Id = f.Id
    LEFT JOIN 
        AssetTypeMapping atm ON fs.AssetType = atm.AssetType
    WHERE 
        fp.Name NOT LIKE '%DO NOT%'
),
CalculatedRanges AS (
    SELECT
        FeePackageName,
        AuthorisedDate,
        CreatedDate,
        ActualisationFrequency,
        CalculationMethod,
        FeeName,
        AssetType,
        AssetTypeName,
        EndValue,
        AnnualisedFeeInPercentage,
        COALESCE(LAG(EndValue) OVER (PARTITION BY FeePackageName, AssetTypeName ORDER BY RowNum), 0) AS StartValue,
        ROW_NUMBER() OVER (PARTITION BY FeePackageName, AssetTypeName ORDER BY RowNum) AS RowNum2
    FROM
        RankedFeeScales
)
SELECT
    FeePackageName,
    AuthorisedDate,
    CreatedDate,
    ActualisationFrequency,
    CalculationMethod,
    FeeName,
    AssetType,
    AssetTypeName,
    CASE 
        WHEN EndValue IS NULL AND AssetTypeName = 'Unitised' THEN 0
        WHEN EndValue IS NULL THEN (SELECT MAX(EndValue) FROM RankedFeeScales WHERE FeePackageName = cr.FeePackageName AND AssetTypeName = cr.AssetTypeName)
        ELSE StartValue 
    END AS StartValue,
    EndValue,
    AnnualisedFeeInPercentage
FROM
    CalculatedRanges cr
	WHERE
	1=1
	AND FeePackageName = 'Craigs KS Fee Schedule'
	AND FeeName = 'Management Fee'
ORDER BY
    FeePackageName, 
	AssetTypeName,
	StartValue;

Condition

Description

Condition 1: Do Not Use Package and Closed

Special handling for portfolios with "Do Not Use" package and closed status

  • FeePackageName LIKE '%Do Not Use%' <br> - PortfolioClosedDate IS NOT NULL <br> - PortfolioStatus <> 'Active'

Sub-condition 1.1: Normal Calculation

Calculate normally before the closed date

  • Check if StartDate < PortfolioClosedDate AND EndDate > PortfolioClosedDate <br> - Asset type-specific calculations

Sub-condition 1.2: Shares

Calculate fee for shares

  • ValuationNZD > StartValue AND (ValuationNZD <= EndValue OR EndValue IS NULL) <br> - Math: ((ValuationNZD - StartValue) * AnnualisedFeeInPercentage / 100 / 365)

Sub-condition 1.3: Cash Call Account

Calculate fee for cash call account

  • ValuationNZD > 10000 <br> - Math: (ValuationNZD - 10000) * (AnnualisedFeeInPercentage / 100) / 365

Sub-condition 1.4: Unitised

Calculate fee for unitised assets

  • ValuationNZD > 0 <br> - Math: ValuationNZD * (AnnualisedFeeInPercentage / 100) / 365

Condition 2: Do Not Use but Still Open

Handle portfolios in "Do Not Use" package but still open

  • Stick to normal calculations without special conditions

Condition 3: Active Portfolios

Logic for active portfolios with normal package

  • Asset type-specific calculations - Similar logic as sub-conditions above
