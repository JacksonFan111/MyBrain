FROM NZXWT_Holdings
WHERE Portfolio IN (SSD_GEORGE, SSD_IAINW, ..., SSD_TOMW)
  AND Date BETWEEN '2024-01-01' AND '2024-12-31'
GROUP BY Asset
ORDER BY Total_Holdings DESC
LIMIT 3;

  AND Portfolio IN (SSD_GEORGE, ..., SSD_TOMW)
  AND Date BETWEEN '2024-01-01' AND '2024-12-31'
GROUP BY Asset
ORDER BY Total_Buys DESC
LIMIT 3;
